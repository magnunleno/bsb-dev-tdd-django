  • Nomenclatura varia de acordo com a linguagem e o framework de testes;
  • Spies

    • "Substituição" de um método ou função simples a fim de evitar uma "comuniação externa";

# cliente.py
import json
from datetime import date, datetime


def fmt_datetime(dt):
    if isinstance(dt, date):
        dt = datetime(
          dt.year, dt.month, dt.day
        )
    return dt.isoformat()


class Cliente:
    def __init__(self, id=None, nome=None,
                 data_nascimento=None):
        self.id = id
        self.nome = nome
        self.data_nascimento = data_nascimento

    def to_json(self):
        raw = {
            'id': self.id,
            'nome': self.nome,
            'data_nascimento': fmt_datetime(
              self.data_nascimento
            ),
        }
        return json.dumps(raw)


import json

from unittest.mock import patch
from datetime import date, datetime

import cliente


def test_cliente_to_json():
    c = cliente.Cliente(id=1, nome="Fulano", data_nascimento=date(2018, 1, 1))

    with patch.object(cliente, 'datetime', wraps=cliente.datetime) as mock:
        raw = json.loads(c.to_json())
        assert mock.called
        assert mock.call_args[0] == (2018, 1, 1)
        assert len(mock.mock_calls) == 1

    assert raw['id'] == 1
    assert raw['nome'] == "Fulano"
    assert raw['data_nascimento'] == "2018-01-01T00:00:00"

    c.data_nascimento = datetime(2018, 1, 1, 12, 00, 10)
    with patch.object(cliente, 'datetime', wraps=cliente.datetime) as mock:
        raw = json.loads(c.to_json())
        assert mock.called is False

    assert raw['id'] == 1
    assert raw['nome'] == "Fulano"
    assert raw['data_nascimento'] == "2018-01-01T12:00:10"
