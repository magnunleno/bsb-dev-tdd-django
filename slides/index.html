Spies

  • Registra seus argumentos, quantidade de vezes que foi invocada, retorno...

Spies


# utils.py
import datetime


def fmt_datetime(dt):
    if type(dt) == date:
        dt = datetime.datetime(
          dt.year,
          dt.month,
          dt.day,
        )
    return dt.isoformat()


# test_utils.py
from unittest.mock import patch
from datetime import date, datetime

import utils


def test_fmt_datetime():
    with patch.object(utils, 'datetime', wraps=utils.datetime) as spie:
        fmt = utils.fmt_datetime(
          date(2018, 1, 1)
        )
        assert spie.called
        assert spie.call_args[0] == (2018, 1, 1)
        assert len(spie.mock_calls) == 1
        assert fmt == "2018-01-01T00:00:00"

        spie.reset_mock()

        fmt = utils.fmt_datetime(
          datetime(2018, 1, 1, 2, 3, 4)
        )
        assert not spie.called
        assert fmt == "2018-01-01T02:03:04"

Stubs


# utils.py
import os

def apaga_arquivo(caminho):
    if os.path.exists(caminho):
        os.remove(caminho)
        return True
    return False


# test_utils.py
import utils
from unittest.mock import patch

@patch('utils.os.remove', return_value=True)
def test_apaga_arquivo(mock):
    with patch('utils.os.path.exists', return_value=True):
        success = utils.apaga_arquivo('/phony/file')
        assert success

    with patch('utils.os.path.exists', return_value=False):
        success = utils.apaga_arquivo('/phony/file')
        assert not success

Mock

  • Stub de um "objeto complexo";
  • Possui funcionalidades de spies e de stubs

Mock


class Cliente:
    def __init__(self, id=None, nome=None):
        self.id = id
        self.nome = nome

    @classmethod
    def use(kls, api=None):
        if api:
            kls.api = api

    @classmethod
    def por_id(kls, id):
        data = kls.api.fetch(id)
        if data is None:
            return None
        return kls(**data)


from unittest.mock import MagicMock

def test_cliente_api():
    mock = MagicMock()
    utils.Cliente.use(api=mock)

    mock.fetch.return_value = {"id": 1, "nome": "Fulano"}
    c = utils.Cliente.por_id(1)
    assert mock.fetch.call_count == 1
    assert mock.fetch.call_args[0] == (1,)
    assert c.id == 1
    assert c.nome == "Fulano"

    mock.fetch.reset_mock()
    mock.fetch.return_value = None
    c = utils.Cliente.por_id(3)
    assert c is None
    assert mock.fetch.call_count == 1
    assert mock.fetch.call_args[0] == (3,)
